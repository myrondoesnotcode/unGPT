  <script>

    const $ = (sel) => document.querySelector(sel);

    const input = $('#input');

    const output = $('#output');

    const countEl = $('#count');


    function ungpt(text){

      if(!text) return '';

      // Replace em dash and en dash with '-'

      let replaced = text

        .replace(/[\\u2014\\u2015]/g, '-') // em dash & horizontal bar

        .replace(/[\\u2013]/g, '-')       // en dash

        .replace(/-{2,}/g, '-')          // any sequence of 2+ hyphens

      ;

      // Update count: how many replacements occurred

      const matches = text.match(/[\\u2014\\u2015\\u2013]|-{2,}/g);

      const n = matches ? matches.length : 0;

      countEl.textContent = n;

      return replaced;

    }


    function convert(){

      output.value = ungpt(input.value);

      output.focus();

      output.select();

    }


    // Buttons

    $('#convert').addEventListener('click', convert);

    $('#copy').addEventListener('click', async () => {

      if(!output.value) return;

      try { await navigator.clipboard.writeText(output.value); }

      catch(e) {}

    });

    $('#swap').addEventListener('click', () => {

      const a = input.value, b = output.value;

      input.value = b; output.value = a;

      countEl.textContent = 0;

      input.focus();

    });

    $('#clear').addEventListener('click', () => {

      input.value = ''; output.value=''; countEl.textContent = 0; input.focus();

    });


    // Paste to auto-convert

    input.addEventListener('paste', (e) => {

      requestAnimationFrame(convert);

    });


    // Keyboard shortcut: Cmd/Ctrl + Enter

    document.addEventListener('keydown', (e) => {

      if((e.metaKey || e.ctrlKey) && e.key === 'Enter'){

        e.preventDefault();

        convert();

      }

    });


    // Drag-and-drop .txt files onto the input

    ;['dragenter','dragover','dragleave','drop'].forEach(evt => {

      input.addEventListener(evt, (e)=>{ e.preventDefault(); e.stopPropagation(); });

    });

    input.addEventListener('drop', async (e) => {

      const f = [...e.dataTransfer.files][0];

      if(!f || !f.name.endsWith('.txt')) return;

      const text = await f.text();

      input.value = text;

      convert();

    });


    // Initial focus

    input.focus();

  </script>
